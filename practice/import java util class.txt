import java.util.*;
class Solution{
	public int subArraySum(int[] nums , int k){
		int n = nums.length;
		HashMap<Integer,Integer> hmap = new HashMap<>();
		int sum = 0;
		int max = 0;
		
		for(int i = 0 ; i < n ; i++){
			sum += nums[i];
			if(sum == k){
				max = i+1 ;
			}
			if(hmap.containsKey(sum - k){
				max = Math.max(max, i - hmap.get(sum - k));
			}
			hmap.putIfAbsent(sum,i);
		}
	return max;
	}
	
}
