class Solution{
	public int maxSumByKadane(int[] nums){
		int maxSum  = 0;
		int currSum = 0;
		
		for(int i = 0; i < nums.length; i++){
			currSum+= nums[i];
			maxSum = Math.max(maxSum,currSum);
			
			if(currSum < 0){
				currSum =0;
			}
				
		}
		return maxSum;
	}
	
	public static void main(String args[]){
		int[] nums = {2, 3, 5, -2, 7, -4};
		System.out.println(maxSumByKadane(nums));
	}
}
